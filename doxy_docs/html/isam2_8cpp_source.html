185 void t_associate(vector<Point2> *cone_obs, vector<Pose2> *global_obs_cones,
186 vector<Pose2> *all_cone_est, Pose2 global_odom, vector<float> *m_dist,
187 int lo, int hi)
188 {
189 int i = lo;
190 int landmark_idx = lo % (n_landmarks + 1);
191 int obs_id = (int)(lo / (n_landmarks + 1));
192 //RCLCPP_INFO(logger, "lo: %d | hi: %d", lo, hi);
193
194 while (i < hi && obs_id < cone_obs->size())
195 {
196
197 double bearing = std::atan2(cone_obs->at(obs_id).y(),
198 cone_obs->at(obs_id).x());
199
200 double range = norm2(cone_obs->at(obs_id));
201
202 double global_cone_x = (global_odom.x() + range *
203 cos(bearing + global_odom.theta()));
204
205 double global_cone_y = (global_odom.y() + range *
206 sin(bearing + global_odom.theta()));
207 /* TODO: "valid bit" for whether global pose has already been calculated at obs_id */
208
209 //TODO: move this outside of t_associate, and before threads are spawned
210 global_obs_cones->at(obs_id) = Pose2(global_cone_x, global_cone_y, bearing);
211
212
226 int last_est_for_cur_obs = (obs_id + 1) * (n_landmarks + 1) - 1;
227
228
229 /* calculating the mahalanobis distances */
230 for (; i < last_est_for_cur_obs && i < hi; i++)
231 {
236 Eigen::MatrixXd diff(1, 3);
237 Pose2 prev_est = all_cone_est->at(landmark_idx);
238 diff << global_cone_x - prev_est.x(),
239 global_cone_y - prev_est.y(),
240 1;
241
242 /* calculate the mahalanobis distance */
243 m_dist->at(i) = (diff*isam2.marginalCovariance(L(landmark_idx))
244 * diff.transpose())(0, 0);
245
246 landmark_idx++;
247 }
248
249 if (i == hi)
250 {
251 return;
252 }
253
254 m_dist->at(last_est_for_cur_obs) = M_DIST_TH;
255 i++; /* skip 1 element (reserved for M_DIST_TH) */
256 obs_id++;
257
258 /* new multiple of (n_landmarks+1) means calculating for previous cone/landmark 0 */
259 landmark_idx = 0;
260 }
261
262 }
